Eingansdaten

Treibstoffe und ROF

fuel = 'CH4';
oxid = 'O2(L)';
rof = 3.4; % [-]
% Kontraktionsverhältnis und Düsenhals
d_t = 28.0 * 0.001; % [m]
A_t = pi() / 4 * d_t^2; % [m^2]
acat = (pi() / 4 * 0.05^2) / A_t; % [-]
% Brennkammerdruck
pcc = 60; % [bar]

Berechnung von Dichte und Schallgeschwindigkeit im Düsenhals

ceain(fuel,oxid,rof,pcc,acat);

dos fcea2.exe;
% ! fcea2.exe

C = ceaout;

% Dichte im Düsenhals
index_rhostar = strmatch('RHO',C{1,1}) + 7;
rhostar = str2double(C{1,1}{index_rhostar,1}); % [kg/m^3]

% Schallgeschwindigkeit im Düsenhals
index_a_t = strmatch('SON',C{1,1}) + 4;
a_t = str2double(C{1,1}{index_a_t,1}); % [m/s]

Benötigter Massenstrom

Gesamt

mdot = rhostar * a_t * A_t

% Sauerstoff
mdot_O2 = rof / (1 + rof) * mdot
% Anteil am möglichen Gesamtoxidatormassenstrom am P6
mdot_O2_r = mdot_O2 / 1

% Methan
mdot_CH4 = mdot_O2 / rof
% Anteil am möglichen Gesamtmethanmassenstrom (Haupt- und
% Fensterkühlmassenstrom) am P6
mdot_CH4 = mdot_CH4 / 0.4
